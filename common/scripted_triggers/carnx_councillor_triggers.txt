carnx_can_be_chancellor_trigger = {
	can_be_chancellor_trigger = { COURT_OWNER = $COURT_OWNER$ }
	trigger_if = {
		limit = { carnx_is_slave_trigger = yes }
		$COURT_OWNER$ = {
			carnx_is_slave_employment_enabled_trigger = yes
		}
	}
