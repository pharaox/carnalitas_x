# Called from code once a year for "playable" (count+) characters.
# Root is the character
yearly_playable_pulse = {
	events = {
		carnx_slave.0004 # Piety Loss From Slaves
	}

	random_events = {
		200 = 0
		 50 = carnx_slave.0010 # Slave Converts to Your Faith
	}

	on_actions = {
		delay = { days = { 8 288 } }
		carnx_yearly_playable_pulse
		delay = { days = { 8 188 } }
		carnx_yearly_market
	}
}

carnx_yearly_playable_pulse = {
	trigger = {
		basic_is_valid_for_yearly_events_trigger = yes
	}

	events = {
		carnx_slave.0002 # Sex With a Slave
	}

	random_events = {
		chance_of_no_event = {
			value = 0
			if = {
				limit = { is_at_war = yes }
				add = 50
			}
			if = {
				limit = {
					is_ai = yes
					highest_held_title_tier < tier_duchy
				}
				add = 50
			}
		}

		200 = 0

		 20 = carnx_slave.0003 # Runaway Slave
		100 = carnx_slave.0005 # Freedom for a Slave
		100 = carnx_slave.0009 # The Price of Freedom
		100 = carnx_slave.0011 # The Informant's Bargain
	}
}

carnx_yearly_market = {
	trigger = {
		trigger_if = {
			limit = { is_ai = no }
			has_game_rule = carnx_market_event_frequency_player_1y
		}
		trigger_else = {
			always = no
		}
		basic_is_valid_for_yearly_events_trigger = yes
	}

	events = {
		carnx_slave.0001 # Buy Foreign Slaves
	}
}

# Called from code once every three years for "playable" (count+) characters.

		carnx_eight_year_market

carnx_three_year_market = {
	trigger = {
		trigger_if = {
			limit = { is_ai = no }
			has_game_rule = carnx_market_event_frequency_player_3y
		}
		trigger_else = {
			highest_held_title_tier >= tier_kingdom
		}
		basic_is_valid_for_yearly_events_trigger = yes
	}

	events = {
		carnx_slave.0001 # Buy Foreign Slaves
	}
}

carnx_five_year_market = {

carnx_eight_year_market = {
	trigger = {
		is_ai = yes
		highest_held_title_tier = tier_county
		NOT = { has_character_flag = carnx_eight_year_market_cooldown }
		basic_is_valid_for_yearly_events_trigger = yes
	}

	events = {
		carnx_slave.0001 # Buy Foreign Slaves
	}

	effect = {
		add_character_flag = {
			flag = carnx_eight_year_market_cooldown
			years = 8
		}
	}
}
